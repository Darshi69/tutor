  <script>
    const VOUCHER = 'TOP5PARENTS';
    const DISCOUNT = 0.35;
    const LIMIT = 5;
    let usedCount = 0;

    const prices = { online: 100, offline: 150 };
    const totals = {
      online: document.getElementById('total-online'),
      offline: document.getElementById('total-offline')
    };

    function renderTotals(mult=1){
      totals.online.textContent = '₹' + Math.round(prices.online * mult);
      totals.offline.textContent = '₹' + Math.round(prices.offline * mult);
    }
    renderTotals();

    document.getElementById('apply').addEventListener('click', () => {
      const code = document.getElementById('voucher').value.trim().toUpperCase();
      const msg = document.getElementById('voucher-msg');

      if(code === VOUCHER){
        if(usedCount < LIMIT){
          usedCount++;
          renderTotals(1 - DISCOUNT);
          msg.textContent = `✅ Success! 35% off applied. (${LIMIT - usedCount} uses left)`;
          msg.style.color = 'green';
        } else {
          msg.textContent = '❌ Sorry, this voucher has already been fully used.';
          msg.style.color = 'red';
        }
      } else {
        msg.textContent = '⚠️ Invalid code.';
        msg.style.color = 'darkred';
      }
    });

    document.getElementById('year').textContent = new Date().getFullYear();
  </script>
